ZipPlatformFile相关介绍

列出所有文件

TArray<FString> FoundFiles;
UZipPlatformFileBlueprintFunctionLibrary::FindFiles(this, FoundFiles/*所有子文件*/, TEXT("/Game/test_zip")/*挂载目录*/, TEXT(".txt")/*扩展名*/, true/*是否递归*/);

TArray<FString> FoundDirectories;
UZipPlatformFileBlueprintFunctionLibrary::FindDirectories(this, FoundDirectories/*所有子目录*/, TEXT("/Game/test_zip")/*挂载目录*/);
